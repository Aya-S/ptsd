#Train the chatbot on the intents and strories listed under ptsd_data and output the tained model to ptsd_model
# ptsd_domain defines the universe in which the assistant works in in terms of intents, actions and template responses
rasa train --data ./ptsd_data \
--config ptsd_config.yml \
--domain ptsd_domain.yml \
--out ptsd_models \
--force

#Run the chatbot 
rasa shell --model ptsd_models/

#Train the chatbot interactively 
rasa interactive \
--model ptsd_models \
--data ./ptsd_data \
--config ptsd_config.yml \
--domain ptsd_domain.yml \
--out ptsd_models
